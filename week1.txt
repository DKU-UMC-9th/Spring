[mission 3]
    SELECT m.point, s.name, m.description, um.status
    FROM mission m
    JOIN store s ON m.store_id = s.id
    JOIN user_mission um ON um.mission_id = m.mission_id
    ORDER BY m.mission_id DESC;

[mission 4]
    SELECT m.point, m.description, um.deadline, s.name, s.adress
    FROM mission m
    JOIN store s ON m.store_id = s.id
    JOIN user_mission um ON um.mission_id = m.mission_id
    WHERE um.status = 'n';
    ORDER BY um.deadline ASC
    LIMIT 10 OFFSET 0;
