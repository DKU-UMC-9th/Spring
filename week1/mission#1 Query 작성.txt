SELECT
um.mission_id,
m.name         AS mission_name,
s.name        AS store_name,
m.point        AS mission_point,
um.deadline,
um.status
FROM user_mission um
JOIN mission m ON m.id = um.mission_id
JOIN store   s ON s.id = m.store_id
WHERE um.user_id = :user_id
AND um.status IN ('진행중', '성공')
ORDER BY um.deadline DESC;

AND um.status = '성공';

지역명 + 아직 참여하지 않은 미션 목록
SELECT
:region                          AS region_name,  -- 파라미터로 받은 지역명 그대로 노출
m.id                            AS mission_id,
s.name                           AS store_name,
m.point,
m.deadline
FROM mission m
JOIN store s
ON s.id = m.store_id
LEFT JOIN user_mission um
ON um.mission_id = m.id
AND um.user_id    = :user_id
WHERE s.address LIKE CONCAT(:region, '%')   -- Postgres는 (:region || '%')
AND um.mission_id IS NULL                 -- 아직 참여하지 않은 미션
ORDER BY m.deadline DESC
LIMIT :limit OFFSET (:page - 1) * :limit;
